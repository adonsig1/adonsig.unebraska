\question{
For $S$ a subset of $\bR^n$, define
$\displaystyle S^\perp = \{ x \in \bR^n : x \cdot s = 0 \hbox{ for all } s \in S \}$.
\begin{enumerate}
\item Show that $S^\perp$ is a subspace of $\bR^n$.
\item Show that $S^\perp = (\spann S)^\perp$.
\item Show that $S \subseteq (S^\perp)^\perp$.
\item Show there is a nonempty set $S$ with $S \ne (S^\perp)^\perp$.
\end{enumerate}
}{
