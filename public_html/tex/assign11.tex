\question{Define $f : \bbR \to \bbR$ by $f(x) =\sin(1/x)$ for $x \ne 0$ and
$f(0)=0$.
Show that $f$ has the intermediate value property.

\note of course, $f$ is not continuous.
}{
}
\question{
By directly verifying the $\delta-\epsilon$ condition, show that $f(x)=x^2$ 
is uniformly continuous on $[0,5]$.
}{
}
\question{
\begin{enumerate}
\item Show that if $f : D \to \bbR$ is uniformly continuous on $D$ and
$S \subseteq D$ is a bounded set, then $f$ is a bounded function on $S$,
that is $\{ f(x) : x \in S\}$ is a bounded set.
\bhint assume not and use the Bolzano-Weierstrass Theorem and Theorem 19.4 
(uniformly continuous functions preserve Cauchy sequences).
\item Use part~(a) to give another proof that $f(x)=1/x^2$ is not uniformly
continuous on $(0,1)$.
\end{enumerate}
}{
}
\question{
Suppose that $f : \bbR \to \bbR$ is continuous and periodic, that is, there is $d > 0$ 
so that $f(x+d)=f(x)$ for all $x \in \bbR$.
Show that there are $x_{\hbox{min}}, x_{\hbox{max\vphantom{i}}} \in \bbR$ so that
for all $x \in \bbR$,
\[ f(x_{\hbox{min}}) \le f(x) \le f(x_{\hbox{max\vphantom{i}}}). \]
Further, show that $f$ is uniformly continuous on $\bbR$.
\bhint For the last part, show $f$ is uniformly continuous on $[0,2d]$ and use translation.
}{
}
